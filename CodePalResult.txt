#include <iostream>
#include <string>

using namespace std;

string ai_chatbox(string user_input) {
    /*
    This function takes a string as input and returns a response based on the input.
    
    Parameters:
    user_input (string): The user's input
    
    Returns:
    string: The AI's response
    */
    
    // Convert the user input to lowercase
    transform(user_input.begin(), user_input.end(), user_input.begin(), ::tolower);
    
    // Check for specific keywords in the user input and return a response accordingly
    if (user_input.find("hello") != string::npos || user_input.find("hi") != string::npos) {
        return "Hello! How can I assist you today?";
    }
    else if (user_input.find("how are you") != string::npos) {
        return "I'm just a computer program, so I don't have feelings. But thank you for asking!";
    }
    else if (user_input.find("what is your name") != string::npos) {
        return "My name is CodePal. How can I help you today?";
    }
    else {
        return "I'm sorry, I didn't understand your question. Can you please rephrase it?";
    }
}

int main() {
    // Test the ai_chatbox function
    string user_input;
    cout << "Welcome to the AI chatbox. How can I assist you today?" << endl;
    while (true) {
        getline(cin, user_input);
        if (user_input == "exit") {
            break;
        }
        cout << ai_chatbox(user_input) << endl;
    }
    return 0;
}
